25  import java.util.Properties;
26  
27  import javax.validation.constraints.NotNull;
28  
29  /**
30   * Utility class for I18N context tests.
31   *
32   * @author <a href="https://github.com/ihernaez">(w) Iker Hernaez</a>
33   * @version 1.0, 2024-08
34   * @since 0.1
35   */
36  public final class ContextTestUtils {
37  
38      /**
39       * Private constructor.
40       */
41      private ContextTestUtils() {
42          // Utility class
43      }
44  
45      /**
46       * Resets the I18N configuration and context provider registry.
47       */
48      public static void reset() {
49          I18nConfiguration.reset();
50          I18nContextProvider.Registry.reset();
51      }
52  
53      /**
54       * Sets the I18N configuration for current thread context class loader.
55       * 
56       * @param config The I18N configuration.
57       */
58      public static void setConfiguration(
59              final @NotNull Properties config) {
60          setConfiguration(Thread.currentThread().getContextClassLoader(), config);
61      }
62  
63      /**
64       * Sets the I18N configuration for the specified class loader.
65       * 
66       * @param cl The class loader.
67       * @param config The I18N configuration.
68       */
69      public static void setConfiguration(
70              final @NotNull ClassLoader cl,
71              final @NotNull Properties config) {
72          I18nConfiguration.set(cl, config);
73      }
74  
75      /**
76       * Sets the I18N context provider for current thread context class loader.
77       * 
78       * @param provider The I18N context provider.
79       */
80      public static void setProvider(
81              final @NotNull I18nContextProvider provider) {
82          setProvider(Thread.currentThread().getContextClassLoader(), provider);
83      }
84  
85      /**
86       * Sets the I18N context provider for the specified class loader.
87       * 
88       * @param cl The class loader.
89       * @param provider The I18N context provider.
90       */
91      public static void setProvider(
92              final @NotNull ClassLoader cl,
93              final @NotNull I18nContextProvider provider) {
94          I18nContextProvider.Registry.set(cl, provider);
95      }
96  }
