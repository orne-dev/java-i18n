25  import org.apiguardian.api.API;
26  import org.apiguardian.api.API.Status;
27  
28  import org.springframework.context.annotation.Bean;
29  import org.springframework.context.annotation.Configuration;
30  
31  /**
32   * I18N context provider configuration for Spring Web.
33   * Configures a listener for clearing I18N context after handling the HTTP
34   * requests.
35   * 
36   * @author <a href="https://github.com/ihernaez">(w) Iker Hernaez</a>
37   * @version 1.0, 2023-05
38   * @see I18nSpringConfiguration
39   * @see I18nSpringWebContextClearer
40   * @since 0.1
41   */
42  @API(status=Status.STABLE, since="0.1")
43  @Configuration
44  public class I18nSpringWebConfiguration extends I18nSpringConfiguration {
45  
46      /** The I18N context clearer  */
47      public static final String CLEARER_BEAN_NAME =
48              "i18nSpringWebContextCleaner";
49  
50      /**
51       * Creates a new instance.
52       */
53      public I18nSpringWebConfiguration() {
54          super();
55      }
56  
57      /**
58       * Creates the Spring listener for clearing the I18N context after
59       * handling the HTTP requests.
60       * 
61       * @return The Spring listener.
62       */
63      @Bean(name=CLEARER_BEAN_NAME)
64      public I18nSpringWebContextClearer orneBeansI18nContextClearer() {
65          return new I18nSpringWebContextClearer();
66      }
67  }
