25  import java.lang.annotation.Annotation;
26  
27  import javax.validation.ConstraintValidator;
28  import javax.validation.ConstraintValidatorContext;
29  import javax.validation.constraints.NotNull;
30  
31  import org.apiguardian.api.API;
32  import org.apiguardian.api.API.Status;
33  
34  import dev.orne.i18n.I18nString;
35  import dev.orne.i18n.I18nStringMap;
36  
37  /**
38   * Abstract implementation of Javax Bean Validation
39   * {@code ConstraintValidator} for {@code I18nString} instances.
40   * 
41   * @author <a href="https://github.com/ihernaez">(w) Iker Hernaez</a>
42   * @version 1.0, 2021-02
43   * @param <T> The constraint type to validate
44   * @see ConstraintValidator
45   * @see I18nString
46   * @since 0.1
47   */
48  @API(status=Status.STABLE, since="0.1")
49  public abstract class AbstractValidatorForI18nString<T extends Annotation>
50  implements ConstraintValidator<T, I18nString> {
51  
52      /**
53       * Creates a new instance.
54       */
55      protected AbstractValidatorForI18nString() {
56          super();
57      }
58  
59      /**
60       * {@inheritDoc}
61       */
62      @Override
63      public boolean isValid(
64              final I18nString value,
65              final @NotNull ConstraintValidatorContext context) {
66          if ( value == null ) {
67              return true;
68          }
69          boolean valid = true;
70          if (value instanceof I18nStringMap) {
71              final I18nStringMap map = (I18nStringMap) value;
72              valid = isTextValid(map.getDefaultText());
73              if (valid) {
74                  for (final String translation : map.getI18n().values()) {
75                      valid = isTextValid(translation);
76                      if (!valid) {
77                          break;
78                      }
79                  }
80              }
81          } else {
82              valid = isTextValid(value.get());
83          }
84          return valid;
85      }
86  
87      /**
88       * Validates if the text is valid.
89       * 
90       * @param text The text to validate
91       * @return If the text is valid
92       */
93      protected abstract boolean isTextValid(
94              @NotNull String text);
95  }
