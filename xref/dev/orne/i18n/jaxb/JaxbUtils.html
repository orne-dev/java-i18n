39  /**
40   * Utility class for Jakarta XML Binding (JAXB)
41   * 
42   * @author <a href="https://github.com/ihernaez">(w) Iker Hernaez</a>
43   * @version 1.0, 2021-10
44   * @since 0.1
45   */
46  @API(status=Status.INTERNAL, since="0.1")
47  public final class JaxbUtils {
48  
49      /**
50       * Private constructor.
51       */
52      private JaxbUtils() {
53          // Utility class
54      }
55  
56      /**
57       * Converts the specified {@code I18nString} to {@code XmlI18NString}
58       * containing only the current language translation.
59       * 
60       * @param value The {@code I18nString} to convert
61       * @return The {@code XmlI18NString} to marshall
62       */
63      public static XmlI18nString toXml(
64              final I18nString value) {
65          final XmlI18nString result;
66          if (value == null) {
67              result = null;
68          } else {
69              result = new XmlI18nString();
70              result.getContent().add(value.get());
71          }
72          return result;
73      }
74  
75      /**
76       * Converts the specified {@code I18nString} to {@code XmlI18NString}
77       * containing all the available translations.
78       * Calls {@code I18nString.asMap()} to retrieve the full translations
79       * version.
80       * 
81       * @param value The {@code I18nString} to convert
82       * @return The {@code XmlI18NString} to marshall
83       */
84      public static XmlI18nString toFullXml(
85              final I18nString value) {
86          final XmlI18nString result;
87          if (value == null) {
88              result = null;
89          } else {
90              final I18nStringMap map = value.asMap();
91              result = new XmlI18nString();
92              result.getContent().add(map.getDefaultText());
93              for (final Map.Entry<String,String> entry : map.getI18n().entrySet()) {
94                  final XmlI18nStringTranslation translation = new XmlI18nStringTranslation();
95                  translation.setLang(entry.getKey());
96                  translation.setValue(entry.getValue());
97                  result.getContent().add(translation);
98              }
99          }
100         return result;
101     }
102 
103     /**
104      * Converts the specified {@code XmlI18NString} to {@code I18nString}.
105      * If {@code XmlI18NString} contains {@code XmlI18NStringTranslation}
106      * instances a {@code I18nStringMap} is returned.
107      * Otherwise a {@code I18nFixedString} is returned.
108      * 
109      * @param value The {@code XmlI18NString} to convert
110      * @return The equivalent {@code I18nString}
111      */
112     public static @NotNull I18nString fromXml(
113             final @NotNull XmlI18nString value) {
114         final StringBuilder buffer = new StringBuilder();
115         final List<XmlI18nStringTranslation> translations = new ArrayList<>();
116         for (final Serializable part : value.getContent()) {
117             if (part instanceof XmlI18nStringTranslation) {
118                 translations.add((XmlI18nStringTranslation) part);
119             } else {
120                 buffer.append(part.toString());
121             }
122         }
123         final String text = buffer.toString().trim();
124         final I18nString result;
125         if (translations.isEmpty()) {
126             result = I18nFixedString.from(text);
127         } else {
128             final I18nStringMap tmp = new I18nStringMap(text);
129             for (final XmlI18nStringTranslation trans : translations) {
130                 tmp.set(
131                         trans.getLang(),
132                         trans.getValue());
133             }
134             result = tmp;
135         }
136         return result;
137     }
