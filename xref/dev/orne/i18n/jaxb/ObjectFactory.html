25  import javax.xml.bind.annotation.XmlRegistry;
26  
27  import org.apiguardian.api.API;
28  import org.apiguardian.api.API.Status;
29  
30  /**
31   * This object contains factory methods for each Java content interface and
32   * Java element interface in the dev.orne.i18n package.
33   * <p>
34   * An ObjectFactory allows you to programatically construct new instances of
35   * the Java representation for XML content. The Java representation of XML
36   * content can consist of schema derived interfaces and classes representing
37   * the binding of schema type definitions, element declarations and model
38   * groups.
39   * Factory methods for each of these are provided in this class.
40   */
41  @XmlRegistry
42  @API(status=Status.INTERNAL, since="0.1")
43  public class ObjectFactory {
44  
45      /**
46       * Create a new ObjectFactory that can be used to create new instances of
47       * schema derived classes for package: dev.orne.i18n.
48       */
49      public ObjectFactory() {
50          super();
51      }
52  
53      /**
54       * Create an instance of {@link XmlI18nString}.
55       * 
56       * @return The instance of {@link XmlI18nString}.
57       */
58      public XmlI18nString createXmlI18NString() {
59          return new XmlI18nString();
60      }
61  
62      /**
63       * Create an instance of {@link XmlI18nStringTranslation}.
64       * 
65       * @return The instance of {@link XmlI18nStringTranslation}.
66       */
67      public XmlI18nStringTranslation createTranslation() {
68          return new XmlI18nStringTranslation();
69      }
70  }
