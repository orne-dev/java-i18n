25  import javax.validation.constraints.NotNull;
26  
27  import org.apache.commons.lang3.RandomUtils;
28  import org.apiguardian.api.API;
29  import org.apiguardian.api.API.Status;
30  
31  import dev.orne.test.rnd.Generators;
32  import dev.orne.test.rnd.Priority;
33  import dev.orne.test.rnd.params.AbstractTypedParameterizableGenerator;
34  import dev.orne.test.rnd.params.StringGenerationParameters;
35  
36  /**
37   * Generator of {@code I18nResourcesString} values.
38   * 
39   * @author <a href="https://github.com/ihernaez">(w) Iker Hernaez</a>
40   * @version 1.0, 2023-10
41   * @since 0.1
42   */
43  @API(status=Status.INTERNAL, since="0.1")
44  @Priority(Priority.NATIVE_GENERATORS)
45  public class I18nResourcesStringGenerator
46  extends AbstractTypedParameterizableGenerator<I18nResourcesString, StringGenerationParameters> {
47  
48      /** The minimum random message codes. */
49      protected static final int MIN_CODES = 1;
50      /** The maximum random message codes. */
51      protected static final int MAX_CODES = 3;
52      /** The minimum random message arguments. */
53      protected static final int MIN_ARGS = 0;
54      /** The maximum random message arguments. */
55      protected static final int MAX_ARGS = 3;
56  
57      /**
58       * Creates a new instance.
59       */
60      public I18nResourcesStringGenerator() {
61          super();
62      }
63  
64      /**
65       * {@inheritDoc}
66       */
67      @Override
68      public @NotNull I18nResourcesString defaultValue(
69              final @NotNull StringGenerationParameters parameters) {
70          return I18nResourcesString
71                  .forDefault(Generators.defaultValue(String.class, parameters))
72                  .withCode(Generators.defaultValue(String.class))
73                  .build();
74      }
75  
76      /**
77       * {@inheritDoc}
78       */
79      @Override
80      public @NotNull I18nResourcesString randomValue(
81              final @NotNull StringGenerationParameters parameters) {
82          final I18nResourcesString.Builder builder = I18nResourcesString
83                  .forDefault(Generators.randomValue(String.class, parameters))
84                  .withCode(Generators.randomValue(String.class));
85          final int codes = RandomUtils.nextInt(MIN_CODES, MAX_CODES + 1);
86          for (int i = 1; i < codes; i++) {
87              builder.withCode(Generators.randomValue(String.class));
88          }
89          if (RandomUtils.nextBoolean()) {
90              builder.ofResources(Generators.randomValue(String.class));
91          }
92          final int args = RandomUtils.nextInt(MIN_ARGS, MAX_ARGS + 1);
93          for (int i = 0; i < args; i++) {
94              builder.withArg(Generators.randomValue(String.class));
95          }
96          return builder.build();
97      }
98  }
